# order #
##订单表##
    
    -orderID: primary_key
    >订单序号
    
    -commodityID: foregin_key(Product)
    >商品序号
    
    -itemNum:
    >商品数
    
    -mailOfbuyer: foregin_key(user)
    >买家邮箱
    
    -mailOfseller: foregin_key(user)
    >卖家邮箱
    
    -deposit:
    >订金
    
    -dealType:
    >交易方式
    
    
# commodity #
##商品表##

    -commodityID: primary_key

    >图片地址
    
    -mailOfseller: foregin_key(user)

    >价格
    
    -description:
    >描述

    >商品名字
